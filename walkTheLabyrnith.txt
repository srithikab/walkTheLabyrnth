void walkTheLabyrinth()
{
    repeat(75)
    {
        movingForward();
    }
    
    
    
}
void movingForward()
{
    while (frontIsClear())
    {
        moveForward();
    }
    turnLeft();
    while(!frontIsClear())
    {
        turnRight();
    }
    
}
